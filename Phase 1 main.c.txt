#include <stdio.h>
#include "process.h"
#include "queue.h"
#include "scheduler.h"

int main() {
    Process processes[3];
    Queue readyQueues[PRIORITY_LEVELS];

    for (int i = 0; i < PRIORITY_LEVELS; i++) {
        initQueue(&readyQueues[i]);
    }

    processes[0] = (Process){0, 3, 0, 5, 5, 2, 0, 0, -1, 0, NEW};
    processes[1] = (Process){1, 1, 0, 4, 4, 2, 0, 0, -1, 0, NEW};
    processes[2] = (Process){2, 3, 0, 3, 3, 2, 0, 0, -1, 0, NEW};

    addToReadyQueue(processes, readyQueues, 0);
    addToReadyQueue(processes, readyQueues, 1);
    addToReadyQueue(processes, readyQueues, 2);

    int pid;

    pid = getNextProcess(readyQueues);
    printf("Scheduled: P%d\n", pid);

    pid = getNextProcess(readyQueues);
    printf("Scheduled: P%d\n", pid);

    pid = getNextProcess(readyQueues);
    printf("Scheduled: P%d\n", pid);

    return 0;
}
